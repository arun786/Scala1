# Scala

	package IfElse
	
	object IfElseDemo {
	  def main(args: Array[String]): Unit = {
	
	    val a = 10
	    val b = 20
	
	    val m = if (a > b) a else b
	    println(m)
	  }
	}

# Java

	package IfElse;
	
	public class IfElseDemoJava {
		public static void main(String[] args) {
	
			int a = 10;
			int b = 20;
	
			int c = a > b ? a : b;
			System.out.println(c);
		}
	}
